class Solution {
    public boolean isPalindrome(String s) {
        StringBuilder str = new StringBuilder();
        for(int i = 0; i < s.length(); i++){
            char ch = s.charAt(i);
            if(ch >= 'A' && ch <= 'Z'){
                str.append((char)(ch + 32));
            }else if(ch >= 'a' && ch <= 'z'){
                str.append(ch);
            }else if(ch >= '0' && ch <= '9'){
                str.append(ch);
            }
        }
        System.out.print(str);

        int i = 0; int j = str.length() - 1;
        while(i < j){
            if(str.charAt(i) != str.charAt(j)){
                return false;
            }
            i++; j--;
        }
        return true;
    }
}
